Blink - LED闪烁

Arduino的"Hello World" - 让板载LED闪烁

// 设置LED引脚
const int ledPin = LED_BUILTIN;  // 大多数Arduino板载LED连接的引脚

void setup() {
  // 初始化数字引脚为输出模式
  pinMode(ledPin, OUTPUT);
}

void loop() {
  digitalWrite(ledPin, HIGH);   // 点亮LED
  delay(1000);                  // 等待1秒
  digitalWrite(ledPin, LOW);    // 熄灭LED
  delay(1000);                  // 再等待1秒
}
硬件: Arduino Uno | 所需组件: 无(使用板载LED)

按钮控制LED

使用按钮控制LED灯的开关

const int buttonPin = 2;    // 按钮连接的引脚
const int ledPin = 13;      // LED连接的引脚
int buttonState = 0;        // 存储按钮状态的变量

void setup() {
  pinMode(ledPin, OUTPUT);      // 初始化LED引脚为输出
  pinMode(buttonPin, INPUT);    // 初始化按钮引脚为输入
}

void loop() {
  // 读取按钮状态
  buttonState = digitalRead(buttonPin);
  
  // 检查按钮是否被按下
  if (buttonState == HIGH) {
    digitalWrite(ledPin, HIGH);   // 点亮LED
  } else {
    digitalWrite(ledPin, LOW);    // 熄灭LED
  }
}
硬件: Arduino Uno | 所需组件: 按钮开关 ×1, LED ×1, 220Ω电阻 ×1, 10kΩ电阻 ×1
添加于: 2023-06-12 标签: 输入, 输出, 按钮

DHT11温湿度传感器

读取DHT11传感器的温度和湿度数据

#include <DHT.h>

#define DHTPIN 2        // DHT数据引脚
#define DHTTYPE DHT11   // DHT 11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  delay(2000);  // 两次读取之间等待2秒
  
  float h = dht.readHumidity();      // 读取湿度(%)
  float t = dht.readTemperature();   // 读取温度(摄氏度)
  
  // 检查读取是否失败
  if (isnan(h) || isnan(t)) {
    Serial.println("无法从DHT传感器读取数据!");
    return;
  }
  
  Serial.print("湿度: ");
  Serial.print(h);
  Serial.print("%\t");
  Serial.print("温度: ");
  Serial.print(t);
  Serial.println("°C");
}
硬件: Arduino Uno | 所需组件: DHT11传感器 ×1, 10kΩ电阻 ×1
添加于: 2023-06-15 标签: 传感器, 温湿度, DHT11

L298N控制直流电机

使用L298N电机驱动模块控制直流电机

// 定义电机控制引脚
const int enA = 9;    // 使能引脚
const int in1 = 8;
const int in2 = 7;

void setup() {
  // 设置所有电机控制引脚为输出
  pinMode(enA, OUTPUT);
  pinMode(in1, OUTPUT);
  pinMode(in2, OUTPUT);
  
  // 初始停止电机
  digitalWrite(in1, LOW);
  digitalWrite(in2, LOW);
}

void loop() {
  // 电机正转
  digitalWrite(in1, HIGH);
  digitalWrite(in2, LOW);
  analogWrite(enA, 200);  // 设置速度(0-255)
  delay(2000);
  
  // 电机停止
  digitalWrite(in1, LOW);
  digitalWrite(in2, LOW);
  delay(1000);
  
  // 电机反转
  digitalWrite(in1, LOW);
  digitalWrite(in2, HIGH);
  analogWrite(enA, 200);  // 设置速度(0-255)
  delay(2000);
  
  // 电机停止
  digitalWrite(in1, LOW);
  digitalWrite(in2, LOW);
  delay(1000);
}
硬件: Arduino Uno | 所需组件: L298N电机驱动模块 ×1, 直流电机 ×1, 外部电源 ×1
