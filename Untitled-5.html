// 常量定义
const char CORRECT_PASSWORD[] = "7355608";  // 正确密码
const byte PASSWORD_LENGTH = 7;            // 密码长度
const byte LED_PIN_1 = 12;                 // LED引脚1
const byte LED_PIN_2 = 13;                 // LED引脚2
const byte RELAY_PIN = 10;                 // 继电器引脚(未使用)

// LCD初始化
LiquidCrystal_I2C lcd(0x27, 16, 2);

// 密码输入缓冲区
char inputBuffer[PASSWORD_LENGTH + 1]; // +1 for null terminator
byte inputIndex = 0;

// 键盘配置
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {8, 9, 10, 11};
byte colPins[COLS] = {4, 5, 6, 7};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

void setup() {
  // 初始化LCD
  lcd.init();
  lcd.backlight();
  
  // 初始化串口
  Serial.begin(9600);
  
  // 设置引脚模式
  pinMode(LED_PIN_1, OUTPUT);
  pinMode(LED_PIN_2, OUTPUT);
  pinMode(RELAY_PIN, OUTPUT);
  
  // 初始显示
  showInitialMessage();
}

void loop() {
  char key = keypad.getKey();
  
  if (key) {
    handleKeyPress(key);
  }
}

// 显示初始消息
void showInitialMessage() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("C4 no Password");
  lcd.setCursor(6, 1);
  lcd.print("*******");
}

// 处理按键输入
void handleKeyPress(char key) {
  if (inputIndex < PASSWORD_LENGTH) {
    // 显示按键并存储
    lcd.setCursor(inputIndex + 5, 1);
    lcd.print(key);
    inputBuffer[inputIndex] = key;
    inputIndex++;
    delay(200); // 适当延迟防止按键抖动
  } else {
    // 密码输入完成，进行验证
    checkPassword();
    resetInput();
  }
}

// 验证密码
void checkPassword() {
  lcd.clear();
  
  if (strncmp(inputBuffer, CORRECT_PASSWORD, PASSWORD_LENGTH) == 0) {
    // 密码正确，触发倒计时
    startCountdown();
    showBoomMessage();
  } else {
    // 密码错误
    showErrorMessage();
  }
}

// 重置输入
void resetInput() {
  inputIndex = 0;
  memset(inputBuffer, 0, sizeof(inputBuffer));
  showInitialMessage();
}

// 显示错误消息
void showErrorMessage() {
  lcd.setCursor(0, 0);
  lcd.print("password error");
  delay(3000);
}

// 显示爆炸消息
void showBoomMessage() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("boom!!!");
  lcd.setCursor(2, 1);
  lcd.print("******");
}

// 倒计时效果
void startCountdown() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Bomb activation");
  lcd.setCursor(6, 1);
  lcd.print("******");
  
  // 倒计时阶段1: 慢速闪烁
  blinkSequence(500, 10);
  
  // 倒计时阶段2: 中速闪烁
  blinkSequence(400, 10);
  
  // 倒计时阶段3: 快速闪烁
  blinkSequence(300, 10);
  
  // 倒计时阶段4: 极快闪烁
  blinkSequence(200, 10);
  
  // 最后急促闪烁
  blinkSequence(100, 4);
  blinkSequence(50, 4);
}

// LED闪烁序列
void blinkSequence(int interval, int times) {
  for (int i = 0; i < times; i++) {
    digitalWrite(LED_PIN_1, HIGH);
    digitalWrite(LED_PIN_2, HIGH);
    delay(interval);
    digitalWrite(LED_PIN_1, LOW);
    digitalWrite(LED_PIN_2, LOW);
    delay(interval);
  }
}

功能说明

密码输入功能

通过4×4矩阵键盘输入7位密码(7355608)，密码正确后触发倒计时效果。

LCD显示反馈

1602 LCD显示屏实时显示输入状态、错误提示和倒计时动画。

LED模拟效果

通过LED闪烁模拟炸弹倒计时和爆炸效果，逐渐加快闪烁频率。
